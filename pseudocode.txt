=====================================================
ESTRUCTURA DE DATOS PRINCIPALES:
=====================================================

courses_list: Lista que contiene diccionarios con formato:
    {
        "nombre": "Nombre del curso",
        "nota": valor_numerico
    }

pila_historial: Lista que funciona como pila LIFO para almacenar:
    {
        "accion": "eliminado" | "actualizado",
        "curso": "Nombre del curso",
        "nota_anterior": valor (si aplica),
        "nota_nueva": valor (si aplica)
    }

cola_revisiones: Lista que funciona como cola FIFO para:
    ["nombre_curso1", "nombre_curso2", ...]


1. registrar_curso()
   - <VALIDAR_NOMBRE_NO_VACIO>
   - <VALIDAR_NOTA_ENTRE_0_Y_100>
   - <AGREGAR_A_courses_list>

2. mostrar_todos_cursos()
   - <VERIFICAR_LISTA_NO_VACIA>
   - <RECORRER_E_IMPRIMIR_CURSOS>

3. calcular_promedio_general()
   - <SUMAR_TODAS_LAS_NOTAS>
   - <DIVIDIR_ENTRE_CANTIDAD_CURSOS>
   - <REDONDEAR_A_DOS_DECIMALES>

4. contar_aprobados_reprobados()
   - <RECORRER_LISTA_CONTANDO_SEGUN_NOTA_MAYOR_O_IGUAL_60>

5. buscar_curso_lineal()
   - <RECORRER_SECUENCIALMENTE_COMPARANDO_NOMBRES>

6. actualizar_nota_curso()
   - <BUSCAR_CURSO_POR_NOMBRE>
   - <VALIDAR_NUEVA_NOTA>
   - <ACTUALIZAR_Y_AGREGAR_AL_HISTORIAL>

7. eliminar_curso()
   - <BUSCAR_CURSO_POR_NOMBRE>
   - <SOLICITAR_CONFIRMACION>
   - <ELIMINAR_Y_AGREGAR_AL_HISTORIAL>

8. ordenar_por_nota_burbuja()
   - <IMPLEMENTAR_ALGORITMO_BURBUJA_DESCENDENTE>

9. ordenar_por_nombre_insercion()
   - <IMPLEMENTAR_ALGORITMO_INSERCION_ALFABETICO>

10. buscar_curso_binario()
    - <VERIFICAR_LISTA_ORDENADA_POR_NOMBRE>
    - <IMPLEMENTAR_BUSQUEDA_BINARIA>

11. simular_cola_revisiones()
    - <LLENAR_COLA_CON_NOMBRES_DE_CURSOS>
    - <PROCESAR_COLA_FIFO>

12. mostrar_historial_cambios()
    - <MOSTRAR_PILA_EN_ORDEN_LIFO>
